class User_Friend(models.Model):
    inviter_ID = models.IntegerField()
    receiver_ID = models.IntegerField()
    is_approved = models.BooleanField()


class Mapy(models.Model):
    LokalizacjaID = models.IntegerField()


class Lokalizacje(models.Model):
    LocationName = models.CharField(max_length=40)
    x_cords = models.IntegerField()
    y_cords = models.IntegerField()
    image = models.CharField(max_length=255)
    is_favourite = models.BooleanField()
    Category = models.CharField(max_length=30)


class Plany_Zajec(models.Model):
    LekcjaID = models.IntegerField()


class Lekcje(models.Model):
    PlanID = models.IntegerField()
    NazwaPrzedmiotu = models.CharField(max_length=30)
    Sala = models.CharField(max_length=30)
    Godzina_Rozpoczecia = models.TimeField()
    Godzina_Zakonczenia = models.TimeField()
    Data = models.DateField()


class Filtry(models.Model):
    MapaID = models.IntegerField()
    Category_Name = models.IntegerField()
    is_active = models.IntegerField()


class Kupony(models.Model):
    LokalizacjaID = models.IntegerField()
    is_active = models.IntegerField()
    ilosc_realizacji = models.IntegerField()
    data_realizacji = models.DateField()
